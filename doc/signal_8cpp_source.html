336bool signal::frequency_peakNdtrough(){

352 for(int i = 0; i < least_extrema_num; i++){
353 double minima_subPeriod = 1/(val_minimas.time.at(i + 1) - val_minimas.time.at(i));
354 sumFrequency_minimaBased += minima_subPeriod;
355 _minima_periods.push_back( (val_minimas.time.at(i + 1) - val_minimas.time.at(i)) );
356 }
357
358 double base_frequency_minimaBased = sumFrequency_minimaBased/(least_extrema_num);
359
360 //CHECK that all maxima-to-maxima periods where equal
361 if(_maxima_periods.size() > 1){
362 //START by assuming it is periodic unless its not
363 maxima_periodic = true;
364 cout << base_frequency_maximaBased << "::::::" << base_frequency_minimaBased << endl;
365 for(int i = 0; i < (_maxima_periods.size() - 1); i++){
366 if(!isNear(_maxima_periods.at(i), _maxima_periods.at(i + 1), period_diff_accuracy)){
367 maxima_periodic = false;
368 }
369 }
370 }else{
371 maxima_periodic = true;
372 }
373 if(_minima_periods.size() > 1){
374 minima_periodic = true;
375 for(int i = 0; i < (_minima_periods.size() - 1); i++){
376 if(!isNear(_minima_periods.at(i), _minima_periods.at(i + 1), period_diff_accuracy)){
377 minima_periodic = false;
378 }
379 }
380 }else{
381 minima_periodic = true;
382 }
383
384 //check that all minima-to-minima periods where equal
385
386 //DEDUCE IF THE FUNCTION IS PERIODIC IF time between maximas equals time between minimas and act accordingly
387 if(maxima_periodic && minima_periodic){
388 if( isNear(base_frequency_minimaBased, base_frequency_maximaBased, period_diff_accuracy) ){
389 analytics.base_frequency = (base_frequency_maximaBased + base_frequency_minimaBased)/2;
390 analytics.base_angular_frequency = analytics.base_frequency*M_PI*2;
391 analytics.periods_num = (analytics.timeEnd - analytics.timeStart)*analytics.base_frequency;
392 analytics.periodic_time = 1/analytics.base_frequency;
393 analytics.is_periodic = true;
394
395 unsigned int idx = 0;
396 double t_on = 0;
397 double t_off = 0;
398 double sumDuty = 0;
399 for(idx; idx < least_extrema_num; idx++){
400 t_on = val_minimas.time.at(idx+1) - val_minimas.time.at(idx);

402 sumDuty += t_on/(t_on + t_off);
